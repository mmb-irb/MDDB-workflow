3# Perform the RMSD analysis for each residue 

4# The analysis is carried by pytraj 

5 

6import pytraj as pt 

7 

8from distutils.version import StrictVersion 

9 

10from model_workflow.utils.pyt_spells import get_reduced_pytraj_trajectory 

11from model_workflow.utils.auxiliar import save_json 

12from model_workflow.utils.constants import OUTPUT_RMSD_PERRES_FILENAME 

13from model_workflow.utils.type_hints import * 

14 

15# The pytraj trajectory may be reduced 

16def rmsd_per_residue ( 

17 structure_file : str, 

18 trajectory_file : 'File', 

19 output_directory : str, 

20 structure : 'Structure', 

21 pbc_selection : 'Selection', 

22 snapshots : int, 

23 frames_limit : int): 

24 

25 # Set the main output filepath 

26 output_analysis_filepath = f'{output_directory}/{OUTPUT_RMSD_PERRES_FILENAME}' 

27 

28 # Parse the trajectory intro pytraj 

29 # Reduce it in case it exceeds the frames limit 

30 pt_trajectory, frame_step, frames_count = get_reduced_pytraj_trajectory( 

31 structure_file.path, trajectory_file.path, snapshots, frames_limit) 

32 

33 # We must exclude here PBC residues from the analysis 

34 # PBC residues may jump through boundaries thus having non-sense high RMSD values 

35 # Filter the trajectory with atoms to be included in the analysis 

36 filter_in_selection = structure.invert_selection(pbc_selection) 

37 

38 # Calculate the residue indices of the overall structure remaining in the filtered trajectory 

39 residue_indices = structure.get_selection_residue_indices(filter_in_selection) 

40 # Count the number of residues 

41 residue_count = len(residue_indices) 

42 

43 # Make sure we have more than 1 residue left at this point 

44 # Otherwise it makes not sense running this analysis for a single residue 

45 if residue_count <= 1: 

46 tail = 'no residues at all' if residue_count == 0 else 'one residue only' 

47 print(' The RMSD per residue analysis will be skipped since it would be applied to ' + tail) 

48 return 

49 

50 # Convert the selection to a pytraj mask 

51 pytraj_selection = filter_in_selection.to_pytraj() 

52 # Apply the mask to the trajectory 

53 filtered_pt_trajectory = pt_trajectory[pytraj_selection] 

54 # WARNING: This extra line prevents the error "Segment violation (core dumped)" in some pdbs 

55 # This happens with some random pdbs which pytraj considers to have 0 Mols 

56 # More info: https://github.com/Amber-MD/cpptraj/pull/820 

57 # IMPORTANT: This is critical for pytraj <= 2.0.5 but it makes the code fail for pytraj 2.0.6 

58 if StrictVersion(pt.__version__) <= StrictVersion('2.0.5'): 

59 filtered_pt_trajectory.top.start_new_mol() 

60 

61 # Run the analysis in pytraj 

62 # The result data is a custom pytraj class: pytraj.datasets.datasetlist.DatasetList 

63 # This class has keys but its attributes can not be accessed through the key 

64 # They must be accessed through the index 

65 # IMPORTANT 

66 # Note that the argument 'resrange' may be redundant, since we always target for all residues 

67 # If this argument is not passed then it prints "Error: Range::SetRange(None): Range is -1 for None" 

68 # However, most of the times there is no problem and the analysis runs flawlessly anyway 

69 # We started to have problems with some ions but then we had no results at all with coarse grain 

70 # More info: https://github.com/Amber-MD/pytraj/issues/1580 

71 resrange = f'1-{structure.residue_count + 1}' 

72 data = pt.rmsd_perres(filtered_pt_trajectory, resrange=resrange) 

73 

74 # We remove the first result, which is meant to be the whole rmsd and whose key is 'RMSD_00001' 

75 del data[0] 

76 

77 # Check the expected output number of residues to match with the pytraj results 

78 if residue_count != len(data): 

79 raise ValueError(f'Number of residues in structure ({residue_count}) does not match number of residues in RMSD-perres results ({len(data)})') 

80 

81 # Mine the analysis data 

82 whole_structure_residues_count = len(structure.residues) 

83 rmsd_per_residue = [ None ] * whole_structure_residues_count 

84 for index, residue_data in enumerate(data): 

85 # Get the actual residue index of the current data 

86 residue_index = residue_indices[index] 

87 # Add current data to its corresponding position in the 'per residue' list 

88 rmsd_per_residue[residue_index] = list(residue_data) 

89 

90 # Export the analysis in json format 

91 output_analysis = { 'step': frame_step, 'rmsdpr': rmsd_per_residue } 

92 save_json(output_analysis, output_analysis_filepath)
