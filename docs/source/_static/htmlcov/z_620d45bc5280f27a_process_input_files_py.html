24def process_input_files ( 

25 input_structure_file : 'File', 

26 input_trajectory_files : 'File', 

27 input_topology_file : 'File', 

28 topology_filepath : str, 

29 # Processing parameters 

30 filter_selection : str, 

31 image : bool, 

32 fit : bool, 

33 translation : Tuple, 

34 # Make sure the MD is used only to set values or use its functions, but not to get values 

35 # Values msut be passed separatedly as inputs so the taks can identify when inputs change 

36 self : 'MD', 

37 # Get the task which is calling this function 

38 # Thus we may knwo which inputs have changed compared to previous runs 

39 task : 'Task', 

40): 

41 """Process input files to generate the processed files. 

42 This process corrects and standarizes the topology, the trajectory and the structure.""" 

43 

44 # Make sure we do not enter in a loop 

45 # This may happen when we read/call an output value/file by mistake 

46 if hasattr(self, '_processed'): raise RuntimeError('Looped processing') 

47 self._processed = True 

48 

49 # Set the output filepaths 

50 output_structure_filepath = self.pathify(STRUCTURE_FILENAME) 

51 output_structure_file = File(output_structure_filepath) 

52 output_trajectory_filepath = self.pathify(TRAJECTORY_FILENAME) 

53 output_trajectory_file = File(output_trajectory_filepath) 

54 output_topology_filepath = topology_filepath 

55 output_topology_file = File(output_topology_filepath) if output_topology_filepath else MISSING_TOPOLOGY 

56 

57 # --- FIRST CHECK ----------------------------------------------------------------------- 

58 

59 check_inputs(input_structure_file, input_trajectory_files, input_topology_file) 

60 

61 # --- CONVERTING AND MERGING ------------------------------------------------------------ 

62 

63 # Set the output format for the already converted structure 

64 input_structure_format = input_structure_file.format 

65 output_structure_format = output_structure_file.format 

66 converted_structure_filepath = self.pathify(CONVERTED_STRUCTURE) 

67 # If input structure already matches the output format then avoid the renaming 

68 if input_structure_format == output_structure_format: 

69 converted_structure_filepath = input_structure_file.path 

70 # Set the output file for the already converted structure 

71 converted_structure_file = File(converted_structure_filepath) 

72 # Input trajectories should have all the same format 

73 input_trajectory_formats = set([ trajectory_file.format for trajectory_file in input_trajectory_files ]) 

74 if len(input_trajectory_formats) > 1: 

75 raise InputError('All input trajectory files must have the same format') 

76 # Set the output format for the already converted trajectory 

77 input_trajectories_format = list(input_trajectory_formats)[0] 

78 output_trajectory_format = output_trajectory_file.format 

79 # Set the output file for the already converted trajectory 

80 converted_trajectory_filepath = self.pathify(CONVERTED_TRAJECTORY) 

81 # If input trajectory already matches the output format and is unique then avoid the renaming 

82 if input_trajectories_format == output_trajectory_format and len(input_trajectory_files) == 1: 

83 converted_trajectory_filepath = input_trajectory_files[0].path 

84 converted_trajectory_file = File(converted_trajectory_filepath) 

85 # Join all input trajectory paths 

86 input_trajectory_paths = [ trajectory_file.path for trajectory_file in input_trajectory_files ] 

87 

88 # Set an intermeidate file for the trajectory while it is being converted 

89 # This prevents using an incomplete trajectory in case the workflow is suddenly interrupted while converting 

90 incompleted_converted_trajectory_filepath = self.pathify(INCOMPLETE_PREFIX + CONVERTED_TRAJECTORY) 

91 incompleted_converted_trajectory_file = File(incompleted_converted_trajectory_filepath) 

92 # If there is an incomplete trajectory then remove it 

93 if incompleted_converted_trajectory_file.exists: 

94 incompleted_converted_trajectory_file.remove() 

95 

96 # Convert input structure and trajectories to output structure and trajectory 

97 if not converted_structure_file.exists or not converted_trajectory_file.exists: 

98 print(' * Converting and merging') 

99 convert( 

100 input_structure_filepath = input_structure_file.path, 

101 output_structure_filepath = converted_structure_file.path, 

102 input_trajectory_filepaths = input_trajectory_paths, 

103 output_trajectory_filepath = incompleted_converted_trajectory_file.path, 

104 ) 

105 # Once converted, rename the trajectory file as completed 

106 rename(incompleted_converted_trajectory_file.path, converted_trajectory_file.path) 

107 

108 # Topologies are never converted, but they are kept in their original format 

109 

110 # --- provisional reference structure --- 

111 

112 # Now that we MUST have a PDB file we can set a provisional structure instance 

113 # Note that this structure is not yet corrected so it must be used with care 

114 # Otherwise we could have silent errors 

115 provisional_structure = Structure.from_pdb_file(converted_structure_file.path) 

116 # Now we can set a provisional coarse grain selection 

117 # This selection is useful to avoid problems with CG atom elements 

118 # Since this is proviosonal we will make it silent 

119 provisional_cg_selection = self._set_cg_selection(provisional_structure, verbose=False) 

120 for atom_index in provisional_cg_selection.atom_indices: 

121 provisional_structure.atoms[atom_index].element = CG_ATOM_ELEMENT 

122 

123 # --- FILTERING ATOMS ------------------------------------------------------------ 

124 

125 # Find out if we need to filter 

126 # i.e. check if there is a selection filter and it matches some atoms 

127 must_filter = bool(filter_selection) 

128 

129 # Set output filenames for the already filtered structure and trajectory 

130 # Note that this is the only step affecting topology and thus here we output the definitive topology 

131 filtered_structure_file = File(self.pathify(FILTERED_STRUCTURE)) if must_filter else converted_structure_file 

132 filtered_trajectory_file = File(self.pathify(FILTERED_TRAJECTORY)) if must_filter else converted_trajectory_file 

133 filtered_topology_file = output_topology_file if must_filter else input_topology_file 

134 

135 # Set an intermeidate file for the trajectory while it is being filtered 

136 # This prevents using an incomplete trajectory in case the workflow is suddenly interrupted while filtering 

137 incompleted_filtered_trajectory_filepath = self.pathify(INCOMPLETE_PREFIX + FILTERED_TRAJECTORY) 

138 incompleted_filtered_trajectory_file = File(incompleted_filtered_trajectory_filepath) 

139 # If there is an incomplete trajectory then remove it 

140 if incompleted_filtered_trajectory_file.exists: 

141 incompleted_filtered_trajectory_file.remove() 

142 

143 # Check if any output file is missing 

144 missing_filter_output = not filtered_structure_file.exists or not filtered_trajectory_file.exists 

145 

146 # Check if parameters have changed 

147 # Note that for this specific step only filtering is important 

148 previous_filtered_parameters = self.cache.retrieve(FILTERED) 

149 same_filtered_parameters = previous_filtered_parameters == filter_selection 

150 

151 # Filter atoms in structure, trajectory and topology if required and not done yet 

152 if must_filter and (missing_filter_output or not same_filtered_parameters): 

153 print(' * Filtering atoms') 

154 filter_atoms( 

155 input_structure_file = converted_structure_file, 

156 input_trajectory_file = converted_trajectory_file, 

157 input_topology_file = input_topology_file, # We use input topology 

158 output_structure_file = filtered_structure_file, 

159 output_trajectory_file = incompleted_filtered_trajectory_file, 

160 output_topology_file = filtered_topology_file, # We genereate the definitive topology 

161 reference_structure = provisional_structure, 

162 filter_selection = filter_selection, 

163 ) 

164 # Once filetered, rename the trajectory file as completed 

165 rename(incompleted_filtered_trajectory_file.path, filtered_trajectory_file.path) 

166 # Update the cache 

167 self.cache.update(FILTERED, filter_selection) 

168 

169 # --- provisional reference structure --- 

170 

171 # Now that we have a filtered PDB file we have to update provisional structure instance 

172 # Note that this structure is not yet corrected so it must be used with care 

173 # Otherwise we could have silent errors 

174 provisional_structure = Structure.from_pdb_file(filtered_structure_file.path) 

175 # Again, set the coarse grain atoms 

176 # Since elements may be needed to guess PBC selection we must solve them right before 

177 # Since this is proviosonal we will make it silent 

178 provisional_cg_selection = self._set_cg_selection(provisional_structure, verbose=False) 

179 for atom_index in provisional_cg_selection.atom_indices: 

180 provisional_structure.atoms[atom_index].element = CG_ATOM_ELEMENT 

181 # Also we can set a provisional PBC selection 

182 # This selection is useful both for imaging/fitting and for the correction 

183 # We will make sure that the provisonal and the final PBC selections match 

184 # Since this is proviosonal we will make it silent 

185 provisional_pbc_selection = self._set_pbc_selection(provisional_structure, verbose=False) 

186 

187 # --- IMAGING AND FITTING ------------------------------------------------------------ 

188 

189 # There is no logical way to know if the trajectory is already imaged or it must be imaged 

190 # We rely exclusively in input flags 

191 must_image = image or fit 

192 

193 # Set output filenames for the already filtered structure and trajectory 

194 imaged_structure_file = File(self.pathify(IMAGED_STRUCTURE)) if must_image else filtered_structure_file 

195 imaged_trajectory_file = File(self.pathify(IMAGED_TRAJECTORY)) if must_image else filtered_trajectory_file 

196 

197 # Set an intermeidate file for the trajectory while it is being imaged 

198 # This prevents using an incomplete trajectory in case the workflow is suddenly interrupted while imaging 

199 incompleted_imaged_trajectory_filepath = self.pathify(INCOMPLETE_PREFIX + IMAGED_TRAJECTORY) 

200 incompleted_imaged_trajectory_file = File(incompleted_imaged_trajectory_filepath) 

201 # If there is an incomplete trajectory then remove it 

202 if incompleted_imaged_trajectory_file.exists: 

203 incompleted_imaged_trajectory_file.remove() 

204 

205 # Check if any output file is missing 

206 missing_imaged_output = not imaged_structure_file.exists or not imaged_trajectory_file.exists 

207 

208 # Check if parameters have changed 

209 # Note that for this step the filter parameters is also important 

210 previous_imaged_parameters = self.cache.retrieve(IMAGED) 

211 same_imaged_parameters = previous_imaged_parameters == (image, fit, *translation) 

212 

213 # Image the trajectory if it is required 

214 # i.e. make the trajectory uniform avoiding atom jumps and making molecules to stay whole 

215 # Fit the trajectory by removing the translation and rotation if it is required 

216 if must_image and (missing_imaged_output or not same_imaged_parameters): 

217 print(' * Imaging and fitting') 

218 image_and_fit( 

219 input_structure_file = filtered_structure_file, 

220 input_trajectory_file = filtered_trajectory_file, 

221 input_topology_file = filtered_topology_file, # This is optional if there are no PBC residues 

222 output_structure_file = imaged_structure_file, 

223 output_trajectory_file = incompleted_imaged_trajectory_file, 

224 image = image, 

225 fit = fit, 

226 translation = translation, 

227 structure = provisional_structure, 

228 pbc_selection = provisional_pbc_selection 

229 ) 

230 # Once imaged, rename the trajectory file as completed 

231 rename(incompleted_imaged_trajectory_file.path, imaged_trajectory_file.path) 

232 # Update the cache 

233 self.cache.update(IMAGED, (image, fit, *translation)) 

234 # Update the provisional strucutre coordinates 

235 imaged_structure = Structure.from_pdb_file(imaged_structure_file.path) 

236 imaged_structure_coords = [ atom.coords for atom in imaged_structure.atoms ] 

237 provisional_structure.set_new_coordinates(imaged_structure_coords) 

238 

239 # --- CORRECTING STRUCTURE ------------------------------------------------------------ 

240 

241 # Note that this step, although it is foucsed in the structure, requires also the trajectory 

242 # Also the trajectory may be altered in very rare cases where coordinates must be resorted 

243 

244 # There is no possible reason to not correct the structure 

245 # This is the last step so the output files will be named as the output files of the whole processing 

246 

247 # WARNING: 

248 # For the correcting function we need the number of snapshots and at this point it should not be defined 

249 # Snapshots are calculated by default from the already processed structure and trajectory 

250 # For this reason we can not rely on the public snapshots getter 

251 # We must calculate snapshots here using last step structure and trajectory 

252 snapshots = None 

253 # If we already have a value in the cache then use it, unless the input trajectory has changed 

254 same_trajectory = 'input_trajectory_files' not in task.changed_inputs 

255 if same_trajectory: snapshots = self.cache.retrieve(SNAPSHOTS_FLAG) 

256 # Calculate the new value 

257 if snapshots == None: 

258 snapshots = get_frames_count(imaged_structure_file, imaged_trajectory_file) 

259 # Update the MD snapshots value 

260 self.get_snapshots._set_parent_output(self, snapshots) 

261 # Save the snapshots value in the cache as well 

262 self.cache.update(SNAPSHOTS_FLAG, snapshots) 

263 

264 # WARNING: 

265 # We may need to resort atoms in the structure corrector function 

266 # In such case, bonds and charges must be resorted as well and saved apart to keep values coherent 

267 # Bonds are calculated during the structure corrector but atom charges must be extracted no 

268 charges = get_charges(filtered_topology_file) 

269 self.project.get_charges._set_parent_output(self.project, charges) 

270 

271 print(' * Correcting structure') 

272 

273 # Set output filenames for the already filtered structure and trajectory 

274 corrected_structure_file = File(self.pathify(CORRECTED_STRUCTURE)) 

275 corrected_trajectory_file = File(self.pathify(CORRECTED_TRAJECTORY)) 

276 

277 # Correct the structure 

278 # This function reads and or modifies the following MD variables: 

279 # snapshots, reference_bonds, register, cache, mercy, trust 

280 structure_corrector( 

281 structure = provisional_structure, 

282 input_trajectory_file = imaged_trajectory_file, 

283 input_topology_file = filtered_topology_file, 

284 output_structure_file = corrected_structure_file, 

285 output_trajectory_file = corrected_trajectory_file, 

286 MD = self, 

287 pbc_selection = provisional_pbc_selection, 

288 snapshots = snapshots, 

289 register = self.register, 

290 mercy = self.project.mercy, 

291 trust = self.project.trust 

292 ) 

293 

294 # If the corrected output exists then use it 

295 # Otherwise use the previous step files 

296 # Corrected files are generated only when changes are made in these files 

297 corrected_structure_file = corrected_structure_file if corrected_structure_file.exists else imaged_structure_file 

298 corrected_trajectory_file = corrected_trajectory_file if corrected_trajectory_file.exists else imaged_trajectory_file 

299 

300 # Set for every type of file (structure, trajectory and topology) the input, the last processed step and the output files 

301 input_and_output_files = [ 

302 (input_structure_file, corrected_structure_file, output_structure_file), 

303 (input_trajectory_files[0], corrected_trajectory_file, output_trajectory_file), 

304 (input_topology_file, filtered_topology_file, output_topology_file) 

305 ] 

306 # Set a list of intermediate files 

307 intermediate_files = set([ 

308 converted_structure_file, converted_trajectory_file, 

309 filtered_structure_file, filtered_trajectory_file, 

310 imaged_structure_file, imaged_trajectory_file, 

311 ]) 

312 # Now we must rename files to match the output file 

313 # Processed files remain with some intermediate filename 

314 for input_file, processed_file, output_file in input_and_output_files: 

315 # If the processed file is already the output file then there is nothing to do here 

316 # This means it was already the input file and no changes were made 

317 if processed_file == output_file: 

318 continue 

319 # There is a chance that the input files have not been modified 

320 # This means the input format has already the output format and it is not to be imaged, fitted or corrected 

321 # However we need the output files to exist and we dont want to rename the original ones to conserve them 

322 # In order to not duplicate data, we will setup a symbolic link to the input files with the output filepaths 

323 if processed_file == input_file: 

324 # If output file exists and its the same as the input file, we can not create a symlink from a file to the same file 

325 if output_file.exists: 

326 output_file.remove() 

327 output_file.set_symlink_to(input_file) 

328 # Otherwise rename the last intermediate file as the output file 

329 else: 

330 # In case the processed file is a symlink we must make sure the symlink is not made to a intermediate step 

331 # Intermediate steps will be removed further and thus the symlink would break 

332 # If the symlinks points to the input file there is no problem though 

333 if processed_file.is_symlink(): 

334 target_file = processed_file.get_symlink() 

335 if target_file in intermediate_files: 

336 target_file.rename_to(output_file) 

337 else: 

338 processed_file.rename_to(output_file) 

339 # If the files is not a symlink then simply rename it 

340 else: 

341 processed_file.rename_to(output_file) 

342 

343 # Save the internal variables 

344 self._structure_file = output_structure_file 

345 self._trajectory_file = output_trajectory_file 

346 self.project._topology_file = output_topology_file 

347 

348 # If the input and outpu file have the same name then overwrite input cksums 

349 # Thus we avoid this task to run forever 

350 # DANI: Esto es provisional, hay que prohibir que los inputs se llamen como los outputs 

351 if input_structure_file == output_structure_file: 

352 task.cache_cksums['input_structure_file'] = output_structure_file.get_cksum() 

353 if len(input_trajectory_files) == 1 and input_trajectory_files[0] == output_trajectory_file: 

354 task.cache_cksums['input_trajectory_files'] = get_cksum_id([ output_trajectory_file ]) 

355 if input_topology_file == output_topology_file: 

356 task.cache_cksums['input_topology_file'] = output_topology_file.get_cksum() 

357 

358 # --- Definitive PBC selection --- 

359 

360 # Now that we have the corrected structure we can set the definitive PBC atoms 

361 # Make sure the selection is identical to the provisional selection 

362 if self.pbc_selection != provisional_pbc_selection: 

363 raise InputError('PBC selection is not consistent after correcting the structure. ' 

364 'Please consider using a different PBC selection. ' 

365 'Avoid relying in atom distances or elements to avoid this problem.') 

366 

367 # --- RUNNING FINAL TESTS ------------------------------------------------------------ 

368 

369 # Note that some tests have been run already 

370 # e.g. stable bonds is run in the structure corrector function 

371 

372 # Note that tests here do not modify any file 

373 

374 # Check the trajectory has not sudden jumps 

375 self.is_trajectory_integral() 

376 

377 # Make a final test summary 

378 self.print_tests_summary() 

379 

380 # Issue some warnings if failed or never run tests are skipped 

381 self._issue_required_test_warnings 

382 

383 # --- Cleanup intermediate files 

384 

385 # Set a list of input files to NOT be removed 

386 inputs_files = set([ input_structure_file, *input_trajectory_files, input_topology_file ]) 

387 # We must make sure an intermediate file is not actually an input file before deleting it 

388 removable_files = intermediate_files - inputs_files 

389 # Now delete every removable file 

390 for removable_file in removable_files: 

391 # Note that a broken symlink does not 'exists' 

392 if removable_file.exists or removable_file.is_symlink(): 

393 removable_file.remove()
