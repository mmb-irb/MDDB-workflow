302 # Guess an atom element from its name only 

303 def get_name_suggested_element (self) -> str: 

304 # Get the atom name and its characters length 

305 name = self.name 

306 length = len(name) 

307 next_character = None 

308 for i, character in enumerate(name): 

309 # Get the next character, since element may be formed by 2 letters 

310 if i < length - 1: 

311 next_character = name[i+1] 

312 # If next character is not a string then ignore it 

313 if not next_character.isalpha(): 

314 next_character = None 

315 # Try to get all possible matches between the characters and the supported atoms 

316 # First letter is always caps 

317 character = character.upper() 

318 # First try to match both letters together 

319 if next_character: 

320 # Start with the second letter in caps 

321 next_character = next_character.upper() 

322 both = character + next_character 

323 if both in SUPPORTED_ELEMENTS: 

324 return both 

325 # Continue with the second letter in lowers 

326 next_character = next_character.lower() 

327 both = character + next_character 

328 if both in SUPPORTED_ELEMENTS: 

329 return both 

330 # Finally, try with the first character alone 

331 if character in SUPPORTED_ELEMENTS: 

332 return character 

333 raise InputError(f"Cannot guess atom element from atom name '{name}'")

554 # Get the residue biochemistry classification 

555 # WARNING: Note that this logic will not work in a structure without hydrogens 

556 # Available classifications: 

557 # - protein 

558 # - dna 

559 # - rna 

560 # - carbohydrate 

561 # - fatty 

562 # - steroid 

563 # - ion 

564 # - solvent 

565 # - acetyl 

566 # - amide 

567 # - other 

568 def get_classification (self) -> str: 

569 # Return the internal value, if any 

570 if self._classification: 

571 return self._classification 

572 # If this is a CG residue then we can not classify it 

573 if self.is_cg(): return self.get_classification_by_name() 

574 # If we dont have a value then we must classify the residue 

575 # ------------------------------------------------------------------------------------------------------- 

576 # Ions are single atom residues 

577 if self.atom_count == 1: 

578 self._classification = 'ion' 

579 return self._classification 

580 # ------------------------------------------------------------------------------------------------------- 

581 # At this point we need atoms to have elements fixed 

582 # WARNING: missing elements would result in silent failure to recognize some classifications 

583 if self.structure._fixed_atom_elements == False: 

584 self.structure.fix_atom_elements() 

585 # Solvent is water molecules 

586 # First rely on the residue name 

587 if self.name in STANDARD_SOLVENT_RESIDUE_NAMES: 

588 self._classification = 'solvent' 

589 return self._classification 

590 # It may be water with a not known name 

591 # Literally check if its a molecule with 3 atoms: 2 hydrogens and 1 oxygen 

592 if self.atom_count == 3: 

593 atom_elements = [ atom.element for atom in self.atoms ] 

594 if atom_elements.count('H') == 2 and atom_elements.count('O') == 1: 

595 self._classification = 'solvent' 

596 return self._classification 

597 # ------------------------------------------------------------------------------------------------------- 

598 # Protein definition according to vmd: 

599 # a residue with atoms named C, N, CA, and O 

600 # In our case we accept OC1 and OC2 or OT1 and OT2 instead of O for terminal residues 

601 atom_names = set([ atom.name for atom in self.atoms ]) 

602 if ( all((name in atom_names) for name in ['C', 'N', 'CA']) and ( 

603 'O' in atom_names or { 'OC1', 'OC2' } <= atom_names or { 'OT1', 'OT2' } <= atom_names 

604 )): 

605 self._classification = 'protein' 

606 return self._classification 

607 # ------------------------------------------------------------------------------------------------------- 

608 # Nucleic acids definition according to vmd: 

609 # a residue with atoms named P, O1P, O2P and either O3’, C3’, C4’, C5’, O5’ or O3*, C3*, C4*, C5*, O5* 

610 # Apparently it has been fixed so now a residue does not need to be phosphorylated to be considered nucleic 

611 # LORE: This included the condition "all((name in atom_names) for name in ['P', 'OP1', 'OP2'])" 

612 if (( all((name in atom_names) for name in ["O3'", "C3'", "C4'", "C5'", "O5'"]) or 

613 all((name in atom_names) for name in ["O3*", "C3*", "C4*", "C5*", "O5*"]) )): 

614 # At this point we know it is nucleic 

615 # We must tell the difference between DNA and RNA 

616 if "O2'" in atom_names or "O2*" in atom_names: 

617 self._classification = 'rna' 

618 else: 

619 self._classification = 'dna' 

620 return self._classification 

621 # ------------------------------------------------------------------------------------------------------- 

622 # To define carbohydrates search for rings made of 1 oxygen and 'n' carbons 

623 # WARNING: This logic may fail for some very specific molecules such as furine 

624 # LIMITATION: This logic only aims for cyclical carbohydrates. The linear form of carbohydrates is not yet consider 

625 rings = list(self.find_rings()) 

626 for ring in rings: 

627 ring_elements = [ atom.element for atom in ring ] 

628 # Check the ring has 1 oxygen 

629 oxygen_count = ring_elements.count('O') 

630 if oxygen_count != 1: 

631 continue 

632 # Check the rest are only carbon 

633 carbon_count = ring_elements.count('C') 

634 if carbon_count != len(ring) - 1: 

635 continue 

636 self._classification = 'carbohydrate' 

637 return self._classification 

638 # ------------------------------------------------------------------------------------------------------- 

639 # To define steroids search for 3 6-carbon rings and 1 5-carbon ring (at least) 

640 # WARNING: According to this logic some exotical non-steroid molecules may result in false positives 

641 num_6_carbon_rings = 0 

642 num_5_carbon_rings = 0 

643 for ring in rings: 

644 ring_elements = [ atom.element for atom in ring ] 

645 if any(element != 'C' for element in ring_elements): 

646 continue 

647 ring_lenght = len(ring) 

648 if ring_lenght == 5: 

649 num_5_carbon_rings += 1 

650 if ring_lenght == 6: 

651 num_6_carbon_rings += 1 

652 if num_6_carbon_rings >= 3 and num_5_carbon_rings >= 1: 

653 self._classification = 'steroid' 

654 return self._classification 

655 # ------------------------------------------------------------------------------------------------------- 

656 # To define fatty acids search for a series of carbon atoms connected together (3 at least) 

657 # These carbon atoms must be connected only to hydrogen in addition to 1-2 carbon 

658 # These carbons must not be bonded in a cycle so check atoms not be repeated in the series 

659 bonded_fatty_atoms = [] 

660 def get_bonded_fatty_atoms_recursive (current_atom : Atom, previous_atom : Optional[Atom] = None) -> bool: 

661 # Iterate over the input atom bonded atoms 

662 for bonded_atom in current_atom.get_bonded_atoms(): 

663 # Discard the previous atom to avoid an infinite loop 

664 if bonded_atom == previous_atom: 

665 continue 

666 # If we find an atom which is already in the fatty atoms list then it means we are cycling 

667 if bonded_atom in bonded_fatty_atoms: 

668 return False 

669 # If this is not a fatty atom then discard it 

670 if not bonded_atom.is_fatty_candidate(): 

671 continue 

672 # Add the current bonded fatty atom to the list 

673 bonded_fatty_atoms.append(bonded_atom) 

674 # Find more bonded fatty atoms and add them to list as well 

675 if get_bonded_fatty_atoms_recursive(current_atom=bonded_atom, previous_atom=current_atom) == False: 

676 return False 

677 return True 

678 # Now check all atoms and try to set the series until we find one which works 

679 already_checked_atoms = [] 

680 for atom in self.atoms: 

681 # If we already checked this atom trough the recursive logic then skip it 

682 if atom in already_checked_atoms: 

683 continue 

684 # If the atom is not a fatty candidate then skip it 

685 if not atom.is_fatty_candidate(): 

686 continue 

687 # Now that we have found a suitable candidate atom we start the series 

688 bonded_fatty_atoms = [atom] 

689 if get_bonded_fatty_atoms_recursive(atom) and len(bonded_fatty_atoms) >= 3: 

690 self._classification = 'fatty' 

691 return self._classification 

692 already_checked_atoms += bonded_fatty_atoms 

693 # ------------------------------------------------------------------------------------------------------- 

694 # Check if it is an acetylation capping terminal 

695 if self.name == 'ACE': 

696 self._classification = 'acetyl' 

697 return self._classification 

698 # ------------------------------------------------------------------------------------------------------- 

699 # Check if it is an N-methyl amide capping terminal 

700 if self.name == 'NME': 

701 self._classification = 'amide' 

702 return self._classification 

703 # ------------------------------------------------------------------------------------------------------- 

704 self._classification = 'other' 

705 return self._classification

708 # This is useful for corase grain residues whose atoms may not reflect the real atoms 

709 # WARNING: This logic is very limited and will return "unknown" most of the times 

710 # WARNING: This logic relies mostly in atom names, which may be not standard 

711 def get_classification_by_name (self) -> str: 

712 if self.name in PROTEIN_RESIDUE_NAME_LETTERS: 

713 return 'protein' 

714 if self.name in DNA_RESIDUE_NAME_LETTERS: 

715 return 'dna' 

716 if self.name in RNA_RESIDUE_NAME_LETTERS:

718 if self.name in NUCLEIC_RESIDUE_NAME_LETTERS:

720 if self.name in FATTY_RESIDUE_NAMES:

722 if self.name in STEROID_RESIDUE_NAMES:

724 if self.name in STANDARD_COUNTER_ION_ATOM_NAMES: 

725 return 'ion' 

726 if self.name in STANDARD_SOLVENT_RESIDUE_NAMES: 

727 return 'solvent' 

728 # If we do not know what it is 

729 return 'unknown'

768 # Split this residue in 2 residues and return them in a tuple 

769 # Keep things coherent in the structure (renumerate all residues below this one) 

770 # Note that all residue atoms must be covered by the splits 

771 def split (self, 

772 first_residue_atom_indices : List[int], 

773 second_residue_atom_indices : List[int], 

774 first_residue_name : Optional[str] = None, 

775 second_residue_name : Optional[str] = None, 

776 first_residue_number : Optional[int] = None, 

777 second_residue_number : Optional[int] = None, 

778 first_residue_icode : Optional[str] = None, 

779 second_residue_icode : Optional[str] = None, 

780 ) -> Tuple['Residue', 'Residue']: 

781 # This function is expected to be called in a residue with an already set structure 

782 if not self.structure: 

783 raise InputError('The split function should be called when the residue has an already defined structure') 

784 # Make sure all atoms in the residue are covered between both the first and the second residue 

785 if set(self.atom_indices) != set(first_residue_atom_indices + second_residue_atom_indices): 

786 print('Residue atoms: ' + ', '.join([ str(v) for v in set(self.atom_indices) ])) 

787 print('Covered atoms: ' + ', '.join([ str(v) for v in set(first_residue_atom_indices + second_residue_atom_indices) ])) 

788 raise InputError('All atom indices must be covered between both the first and the second residue') 

789 # Reuse this first residue instead of creating a new one 

790 if first_residue_name: 

791 self.name = first_residue_name 

792 if first_residue_number: 

793 self.number = first_residue_number 

794 if first_residue_icode: 

795 self.icode = first_residue_icode 

796 # Set the new second residue 

797 _second_residue_name = second_residue_name if second_residue_name else self.name 

798 _second_residue_number = second_residue_number if second_residue_number else self.number 

799 _second_residue_icode = second_residue_icode if second_residue_icode else get_next_letter(self.icode) 

800 second_residue = Residue(_second_residue_name, _second_residue_number, _second_residue_icode) 

801 second_residue._structure = self.structure 

802 new_residue_index = self.index + 1 

803 second_residue._index = new_residue_index 

804 # Insert the second residue in the structure residues list right after this residue 

805 self.structure.residues.insert(new_residue_index, second_residue) 

806 # Set the second residue index 

807 # Update the index of all residues which have been shifted after the insertion 

808 for residue_index in range(new_residue_index + 1, len(self.structure.residues)): 

809 residue = self.structure.residues[residue_index] 

810 residue.index = residue_index 

811 # Now transfer atoms from residue 1 to residue 2 as it is specified 

812 # Note that this will automatically update every atom 

813 second_residue.atom_indices = second_residue_atom_indices 

814 # Now add the new residue to the chain 

815 self.chain.add_residue(second_residue)

1098 # Find fragments* in a selection of atoms 

1099 # * A fragment is a selection of colvalently bonded atoms 

1100 # All atoms are searched if no selection is provided 

1101 # WARNING: Note that fragments generated from a specific selection may not match the structure fragments 

1102 # A selection including 2 separated regions of a structure fragment will yield 2 fragments 

1103 def find_fragments (self, selection : Optional['Selection'] = None) -> Generator['Selection', None, None]: 

1104 # If there is no selection we consider all atoms 

1105 if not selection: 

1106 selection = self.select_all() 

1107 # Get/Find covalent bonds between atoms in a new object avoid further corruption (deletion) of the original list 

1108 atom_indexed_covalent_bonds = { atom_index: [ *self.bonds[atom_index] ] for atom_index in selection.atom_indices } 

1109 # Group the connected atoms in "fragments" 

1110 while len(atom_indexed_covalent_bonds) > 0: 

1111 start_atom_index, bonds = next(iter(atom_indexed_covalent_bonds.items())) 

1112 del atom_indexed_covalent_bonds[start_atom_index] 

1113 fragment_atom_indices = [ start_atom_index ] 

1114 while len(bonds) > 0: 

1115 # Get the next bond atom and remove it from the bonds list 

1116 next_atom_index = bonds[0] 

1117 bonds.remove(next_atom_index) 

1118 next_bonds = atom_indexed_covalent_bonds.get(next_atom_index, None) 

1119 # If this atom is out of the selection then skip it 

1120 if next_bonds == None: 

1121 continue 

1122 next_new_bonds = [ bond for bond in next_bonds if bond not in fragment_atom_indices + bonds ] 

1123 bonds += next_new_bonds 

1124 fragment_atom_indices.append(next_atom_index) 

1125 del atom_indexed_covalent_bonds[next_atom_index] 

1126 yield Selection(fragment_atom_indices)

1610 # Fix atom elements by guessing them when missing 

1611 # Set all elements with the first letter upper and the second (if any) lower

1615 def fix_atom_elements (self, trust : bool = True) -> bool: 

1616 modified = False 

1617 added = False 

1618 # Save the wrong guesses for a final report 

1619 # This way we do not crowd the terminal with logs when a lot of atoms are affected

1624 new_element = first_cap_only(atom.element)

1690 # Generate a pdb file content with the current structure 

1691 def generate_pdb (self): 

1692 content = 'REMARK workflow generated pdb file\n' 

1693 for a, atom in enumerate(self.atoms): 

1694 residue = atom.residue 

1695 index = str((a+1) % 100000).rjust(5) 

1696 name = ' ' + atom.name.ljust(3) if len(atom.name) < 4 else atom.name 

1697 residue_name = residue.name.ljust(4) if residue else 'XXX'.ljust(4) 

1698 chain = atom.chain 

1699 chain_name = chain.name if chain.name and len(chain.name) == 1 else 'X' 

1700 residue_number = str(residue.number).rjust(4) if residue else '0'.rjust(4) 

1701 # If residue number is longer than 4 characters then we must parse to hexadecimal 

1702 if len(residue_number) > 4: 

1703 residue_number = hex(residue.number)[2:].rjust(4) 

1704 icode = residue.icode if residue.icode and len(residue.icode) else ' ' 

1705 # Make sure we have atom coordinates 

1706 if atom.coords == None: 

1707 raise InputError('Trying to write a PDB file from a structure with atoms without coordinates') 

1708 x_coord, y_coord, z_coord = [ "{:.3f}".format(coord).rjust(8) for coord in atom.coords ] 

1709 occupancy = '1.00' # Just a placeholder 

1710 temp_factor = '0.00' # Just a placeholder 

1711 element = atom.element.rjust(2) 

1712 atom_line = ('ATOM ' + index + ' ' + name + ' ' + residue_name 

1713 + chain_name + residue_number + icode + ' ' + x_coord + y_coord + z_coord 

1714 + ' ' + occupancy + ' ' + temp_factor + ' ' + element).ljust(80) + '\n' 

1715 content += atom_line 

1716 return content

1950 # Get type of the chain 

1951 def get_selection_classification (self, selection : 'Selection') -> str: 

1952 

1953 # Get selection residues 

1954 selection_residue_indices = self.get_selection_residue_indices(selection) 

1955 

1956 # Inicializar contadores para cada tipo de residuo 

1957 residue_counts = {} 

1958 

1959 # Count the residues of each type 

1960 for residue_index in selection_residue_indices: 

1961 residue = self.residues[residue_index] 

1962 res_class = residue.classification 

1963 if res_class in residue_counts: 

1964 residue_counts[res_class] += 1 

1965 else: 

1966 # If the classification is not recognized, count it as "other" 

1967 residue_counts[res_class] = 1 

1968 

1969 # Count the total number of residues in the selection 

1970 total_residues = sum(residue_counts.values()) 

1971 if total_residues == 0: raise ValueError('Should have residues at this point') 

1972 

1973 # Calculate the proportion of each type of residue 

1974 proportions = { k: v / total_residues for k, v in residue_counts.items() } 

1975 

1976 # If one type of residue dominates, return it 

1977 primary_type = max(proportions, key=proportions.get) 

1978 # We establish a threshold of 80% to consider a chain as a single type 

1979 if proportions[primary_type] >= 0.8: 

1980 return primary_type 

1981 

1982 # Special cases 

1983 relevant_threshold = 0.3 

1984 if proportions.get("dna", 0) > relevant_threshold and proportions.get("rna", 0) > relevant_threshold: 

1985 return "nucleic" 

1986 if proportions.get("carbohydrate", 0) > relevant_threshold and proportions.get("protein", 0) > relevant_threshold: 

1987 return "glycoprotein" 

1988 if proportions.get("fatty", 0) > relevant_threshold and proportions.get("steroid", 0) > relevant_threshold: 

1989 return "lipid" 

1990 

1991 # Any other combinations of different main proportions 

1992 main_proportions = { k: v for k, v in proportions.items() if v > relevant_threshold } 

1993 # Nothing is above the threshold 

1994 if len(main_proportions) == 0: 

1995 return "mix" 

1996 # There is only one thing above threshold 

1997 elif len(main_proportions) == 1: 

1998 return f"{primary_type}/mix" 

1999 # There are two things above threshold 

2000 elif len(main_proportions) == 2: 

2001 other_type = next(key for key in main_proportions.keys() if key != primary_type) 

2002 return f"{primary_type}/{other_type}" 

2003 # There are three things above the threshold (more is not possible) 

2004 else: 

2005 return "mix"

2666 # Find rings in the structure and yield them as they are found 

2667 # LIMITATION: Wrong rings may be found including smaller rings 

2668 # LIMITATION: Use this function to see if a specific ring exists, but not to count rings 

2669 def find_rings (self, selection : Optional[Selection] = None) -> Generator[ List[Atom], None, None ]: 

2670 selected_atom_indices = selection.atom_indices if selection else None 

2671 # We will use a logic which finds long paths of bonded heavy atoms 

2672 # These paths have no 'branches' and thus, when there are 2 possible paths then both paths are explored independently 

2673 # Save already searched atoms and the number of times it has been explored already 

2674 # Note that atoms may be visited as many times as their number of heavy atom bonds -1 

2675 already_searched_atoms = {} 

2676 def find_rings_recursive (atom_path : List[Atom]) -> Generator[ List[Atom], None, None ]: 

2677 # Get the current atom to continue the followup: the last atom 

2678 current_atom = atom_path[-1] # Note that the list MUST always have at least 1 atom 

2679 # Get bonded atoms to continue the path 

2680 followup_atoms = current_atom.get_bonded_atoms() 

2681 # Hydrogens are discarded since they have only 1 bond and thus they can only lead to a dead end 

2682 followup_atoms = [ atom for atom in followup_atoms if atom.element != 'H' ] 

2683 # In case there is a selection, check followup atoms not to be in the selection 

2684 if selected_atom_indices: 

2685 followup_atoms = [ atom for atom in followup_atoms if atom.index in selected_atom_indices ] 

2686 # Check if this atom has been visited already too many times and, if so, stop here 

2687 # Allowed search times is the posible number of rings where this atom can belong 

2688 allowed_search_times = comb(len(followup_atoms), 2) 

2689 already_search_times = already_searched_atoms.get(current_atom, 0) 

2690 if already_search_times == allowed_search_times: 

2691 return 

2692 already_searched_atoms[current_atom] = already_search_times + 1 

2693 # Remove the previous atom to avoid going back 

2694 previous_atom = atom_path[-2] if len(atom_path) > 1 else None 

2695 if previous_atom: 

2696 followup_atoms.remove(previous_atom) 

2697 # Iterate over the following bonded atoms 

2698 for followup_atom in followup_atoms: 

2699 # Check if the followup atom is in the path 

2700 path_index = next(( index for index, atom in enumerate(atom_path) if atom == followup_atom ), None) 

2701 # If so, we have found a ring 

2702 if path_index != None: 

2703 ring = atom_path[path_index:] 

2704 yield ring 

2705 continue 

2706 # Otherwise, keep searching 

2707 new_path = atom_path + [followup_atom] 

2708 for ring in find_rings_recursive(atom_path=new_path): 

2709 yield ring 

2710 # Find a starting atom and run the recursive logic 

2711 candidate_atom_indices = selected_atom_indices if selected_atom_indices else range(self.atom_count) 

2712 for candidate_atom_index in candidate_atom_indices: 

2713 candidate_atom = self.atoms[candidate_atom_index] 

2714 # It must be heavy atom 

2715 if candidate_atom.element == 'H': 

2716 continue 

2717 # It must not be a dead end already 

2718 bonded_atoms = candidate_atom.get_bonded_atoms() 

2719 bonded_atoms = [ atom for atom in bonded_atoms if atom.element != 'H' ] 

2720 # In case there is a selection, check followup atoms not to be in the selection 

2721 if selected_atom_indices: 

2722 bonded_atoms = [ atom for atom in bonded_atoms if atom.index in selected_atom_indices ] 

2723 # Check this atom is not a dead end already 

2724 allowed_search_times = comb(len(bonded_atoms), 2) 

2725 if allowed_search_times < 1: 

2726 continue 

2727 for ring in find_rings_recursive(atom_path=[candidate_atom]): 

2728 yield ring

2758 def find_ptms (self) -> Generator[dict, None, None]:

2823# Calculate the torsion between 4 atoms 

2824# https://stackoverflow.com/questions/20305272/dihedral-torsion-angle-from-four-points-in-cartesian-coordinates-in-python 

2825def calculate_torsion (atom_1 : Atom, atom_2 : Atom, atom_3 : Atom, atom_4 : Atom) -> float: 

2826 p0 = np.array(atom_1.coords) 

2827 p1 = np.array(atom_2.coords) 

2828 p2 = np.array(atom_3.coords) 

2829 p3 = np.array(atom_4.coords) 

2830 # Ge some vectors 

2831 b0 = -1.0*(p1 - p0) 

2832 b1 = p2 - p1 

2833 b2 = p3 - p2 

2834 # normalize b1 so that it does not influence magnitude of vector 

2835 # rejections that come next 

2836 b1 /= np.linalg.norm(b1) 

2837 # vector rejections 

2838 # v = projection of b0 onto plane perpendicular to b1 

2839 # = b0 minus component that aligns with b1 

2840 # w = projection of b2 onto plane perpendicular to b1 

2841 # = b2 minus component that aligns with b1 

2842 v = b0 - np.dot(b0, b1)*b1 

2843 w = b2 - np.dot(b2, b1)*b1 

2844 # angle between v and w in a plane is the torsion angle 

2845 # v and w may not be normalized but that's fine since tan is y/x 

2846 x = np.dot(v, w) 

2847 y = np.dot(np.cross(b1, v), w) 

2848 return float(np.degrees(np.arctan2(y, x)))
